DROP TABLE DEPARTMENTS CASCADE CONSTRAINTS;
DROP TABLE DOCTORS CASCADE CONSTRAINTS;
DROP TABLE PATIENTS CASCADE CONSTRAINTS;
DROP TABLE APPOINTMENTS CASCADE CONSTRAINTS;
DROP TABLE MEDICAL_RECORDS CASCADE CONSTRAINTS;
DROP TABLE MEDICATIONS CASCADE CONSTRAINTS;
DROP TABLE PRESCRIPTIONS CASCADE CONSTRAINTS;

CREATE TABLE DEPARTMENTS (
    DEPARTMENT_ID INT PRIMARY KEY,
    DEPARTMENT_NAME VARCHAR(100) NOT NULL,
    LOCATION_DEPARTMENT VARCHAR(200),
    PHONE_NUMBER VARCHAR(20),
    BUDGET DECIMAL(12,2),
    ESTABLISHED_DATE VARCHAR2(20)
);

CREATE TABLE DOCTORS (
    DOCTOR_ID INT PRIMARY KEY,
    FIRST_NAME VARCHAR(50) NOT NULL,
    LAST_NAME VARCHAR(50) NOT NULL,
    SPECIALIZATION VARCHAR(100),
    LICENSE_NUMBER VARCHAR(20) UNIQUE,
    PHONE_NUMBER VARCHAR(20),
    EMAIL VARCHAR(100),
    HIRE_DATE DATE,
    SALARY DECIMAL(10,2),
    DEPARTMENT_ID INT,
    FOREIGN KEY (DEPARTMENT_ID) REFERENCES DEPARTMENTS(DEPARTMENT_ID)
);

CREATE TABLE PATIENTS (
    PATIENT_ID INT PRIMARY KEY,
    FIRST_NAME VARCHAR(50) NOT NULL,
    LAST_NAME VARCHAR(50) NOT NULL,
    DATE_OF_BIRTH VARCHAR2(20),
    PHONE_NUMBER VARCHAR(20),
    EMAIL VARCHAR(100),
    INSURANCE_NUMBER VARCHAR(50)
);

CREATE TABLE APPOINTMENTS (
    APPOINTMENT_ID INT PRIMARY KEY,
    PATIENT_ID INT NOT NULL,
    DOCTOR_ID INT NOT NULL,
    APPOINTMENT_DATE VARCHAR2(50),
    DURATION_MINUTES INT,
    COST DECIMAL(8,2),
    FOREIGN KEY (PATIENT_ID) REFERENCES PATIENTS(PATIENT_ID),
    FOREIGN KEY (DOCTOR_ID) REFERENCES DOCTORS(DOCTOR_ID)
);

CREATE TABLE MEDICAL_RECORDS (
    RECORD_ID INT PRIMARY KEY,
    PATIENT_ID INT NOT NULL,
    DOCTOR_ID INT NOT NULL,
    VISIT_DATE VARCHAR2(50),
    DIAGNOSIS VARCHAR(500),
    SYMPTOMS VARCHAR(500),
    TREATMENT VARCHAR(500),
    PRESCRIPTION VARCHAR(500),
    FOREIGN KEY (PATIENT_ID) REFERENCES PATIENTS(PATIENT_ID),
    FOREIGN KEY (DOCTOR_ID) REFERENCES DOCTORS(DOCTOR_ID)
);

CREATE TABLE MEDICATIONS (
    MEDICATION_ID INT PRIMARY KEY,
    MEDICATION_NAME VARCHAR(100) NOT NULL,
    MANUFACTURER VARCHAR(100),
    DOSAGE_FORM VARCHAR(50),
    STRENGTH VARCHAR(20),
    UNIT_PRICE DECIMAL(8,2),
    STOCK_QUANTITY INT,
    EXPIRY_DATE VARCHAR2(50)
);

CREATE TABLE PRESCRIPTIONS (
    PRESCRIPTION_ID INT PRIMARY KEY,
    PATIENT_ID INT NOT NULL,
    DOCTOR_ID INT NOT NULL,
    MEDICATION_ID INT NOT NULL,
    PRESCRIPTION_DATE VARCHAR2(50),
    DOSAGE_INSTRUCTIONS VARCHAR(500),
    QUANTITY_PRESCRIBED INT,
    REFILLS_REMAINING INT,
    FOREIGN KEY (PATIENT_ID) REFERENCES PATIENTS(PATIENT_ID),
    FOREIGN KEY (DOCTOR_ID) REFERENCES DOCTORS(DOCTOR_ID),
    FOREIGN KEY (MEDICATION_ID) REFERENCES MEDICATIONS(MEDICATION_ID)
);

-- Rishi Raj Das (Queries)
-- Query 1 – Doctors prescribing the most medications by total quantity
SELECT 
    d.DOCTOR_ID,
    d.FIRST_NAME || ' ' || d.LAST_NAME AS DOCTOR_NAME,
    SUM(p.QUANTITY_PRESCRIBED) AS TOTAL_QUANTITY_PRESCRIBED
FROM DOCTORS d
INNER JOIN PRESCRIPTIONS p ON d.DOCTOR_ID = p.DOCTOR_ID
GROUP BY d.DOCTOR_ID, d.FIRST_NAME, d.LAST_NAME
ORDER BY TOTAL_QUANTITY_PRESCRIBED DESC;

-- Query 2 – Departments whose total salary exceeds their budget
SELECT 
    dept.DEPARTMENT_ID,
    dept.DEPARTMENT_NAME,
    dept.BUDGET,
    SUM(doc.SALARY) AS TOTAL_SALARY
FROM DEPARTMENTS dept
JOIN DOCTORS doc ON dept.DEPARTMENT_ID = doc.DEPARTMENT_ID
GROUP BY dept.DEPARTMENT_ID, dept.DEPARTMENT_NAME, dept.BUDGET
HAVING SUM(doc.SALARY) > dept.BUDGET;

-- Query 3 – Average appointment duration per department
SELECT 
    dept.DEPARTMENT_NAME,
    AVG(app.DURATION_MINUTES) AS AVERAGE_DURATION
FROM DEPARTMENTS dept
JOIN DOCTORS doc ON dept.DEPARTMENT_ID = doc.DEPARTMENT_ID
JOIN APPOINTMENTS app ON doc.DOCTOR_ID = app.DOCTOR_ID
GROUP BY dept.DEPARTMENT_NAME
ORDER BY AVERAGE_DURATION DESC;

-- Query 4 – Medications prescribed but not in stock
SELECT DISTINCT MEDICATION_ID 
FROM PRESCRIPTIONS
MINUS
SELECT DISTINCT MEDICATION_ID 
FROM MEDICATIONS;

-- Query 5 – Patients who have both appointments and prescriptions
SELECT PATIENT_ID 
FROM APPOINTMENTS
INTERSECT
SELECT PATIENT_ID 
FROM PRESCRIPTIONS;

-- Query 6 – Patients without any prescriptions
SELECT 
    p.PATIENT_ID,
    p.FIRST_NAME,
    p.LAST_NAME
FROM PATIENTS p
LEFT JOIN PRESCRIPTIONS pr ON p.PATIENT_ID = pr.PATIENT_ID
WHERE pr.PRESCRIPTION_ID IS NULL;

-- Query 7 – Total cost of appointments vs total unit price of prescriptions per patient
SELECT 
    p.PATIENT_ID,
    'Appointment Cost' AS TYPE,
    SUM(a.COST) AS TOTAL
FROM PATIENTS p
JOIN APPOINTMENTS a ON p.PATIENT_ID = a.PATIENT_ID
GROUP BY p.PATIENT_ID

UNION

SELECT 
    p.PATIENT_ID,
    'Prescription Cost' AS TYPE,
    SUM(m.UNIT_PRICE * pr.QUANTITY_PRESCRIBED) AS TOTAL
FROM PATIENTS p
JOIN PRESCRIPTIONS pr ON p.PATIENT_ID = pr.PATIENT_ID
JOIN MEDICATIONS m ON pr.MEDICATION_ID = m.MEDICATION_ID
GROUP BY p.PATIENT_ID;

-- Query 8 – Doctors in pre-1990 departments with above-average salaries
SELECT 
    d.DOCTOR_ID,
    d.FIRST_NAME || ' ' || d.LAST_NAME AS DOCTOR_NAME,
    d.SALARY,
    dept.DEPARTMENT_NAME,
    dept.ESTABLISHED_DATE
FROM DOCTORS d
JOIN DEPARTMENTS dept ON d.DEPARTMENT_ID = dept.DEPARTMENT_ID
WHERE TO_DATE(dept.ESTABLISHED_DATE, 'YYYY-MM-DD') < TO_DATE('1990-01-01', 'YYYY-MM-DD')
  AND d.SALARY > (SELECT AVG(SALARY) FROM DOCTORS);
